209 static inline void __install_operands(system_t* sys, mmix_processor_t* proc, instr_t* instr)
210 {
211  if(HAS_FLAG(MMIX_RESUMING, proc->state) && proc->rop != MMIX_RESUME_AGAIN) {
212  if(proc->rop == MMIX_RESUME_SET)
213  {
214  instr->op = ORI;
215  instr->y = proc->g[rZ];
216  instr->z = tetra_to_octa(0, 0);
217  instr->exc = ((proc->g[rX] >> 32) & 0xFF00);
218  instr->f = X_is_dest_bit;
219  } else {
220  instr->y = proc->g[rY];
221  instr->z = proc->g[rZ];
222  }
223  } else {
224  if(instr->f & X_is_source_bit) {
225  // Set b from register xx
226  mmix_read_regv(proc, &instr->b, instr->xx);
227  }
228 
229  // Set b from special register
230  if(instr->info->third_operand) {
231  instr->b = proc->g[instr->info->third_operand];
232  }
233 
234  if(instr->f & Z_is_immed_bit) {
235  instr->z = tetra_to_octa(0, instr->zz);
236  } else if(instr->f & Z_is_source_bit) {
237  mmix_read_regv(proc, &instr->z, instr->zz);
238  } else if((instr->op & 0xF0) == SETH) {
239  switch(instr->op & 0x3) {
240  case 0: instr->z |= tetra_to_octa(instr->yz << 16, 0);
241  break;
242  case 1: instr->z |= tetra_to_octa(instr->yz, 0);
243  break;
244  case 2: instr->z |= tetra_to_octa(0, instr->yz << 16);
245  break;
246  case 3: instr->z |= tetra_to_octa(0, instr->yz);
247  break;
248  }
249  instr->y = instr->b;
250  }
251 
252  if(instr->f & Y_is_immed_bit) instr->y = instr->yy;
253  else if(instr->f & Y_is_source_bit) {
254  mmix_read_regv(proc, &instr->y, instr->yy);
255  }
256  }
257 
258  // Install register X as the dest, adjusting the reg stack if required. if L < x < G, then L = x + 1
259  if(instr->f & X_is_dest_bit)
260  {
261  if(instr->xx > proc->G) {
262  instr->x_ptr = &proc->g[instr->xx];
263  } else {
264  while(instr->xx > proc->L) __mmix_lring_push_local(sys, proc);
265  instr->x_ptr = &proc->l[(proc->O + instr->xx) & proc->lmask];
266  }
