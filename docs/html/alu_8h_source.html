256 octa octa_mult(octa y, octa z, octa* aux, bool* overflow)
257 {
258  octa yl = y & 0xFFFFFFFF;
259  octa yh = y >> 32;
260 
261  octa zl = z & 0xFFFFFFFF;
262  octa zh = z >> 32;
263 
264  /*
265  yh yl
266  X zh zl
267  -------------
268  00 LOW PART
269  -------------
270  00
271  10 10 MIDDLE PART
272  + 01
273  -------------
274  01
275  + 11 11 HIGH PART
276  -------------
277  */
278 
279  octa phh = yh * zh, plh = yl * zh;
280  octa phl = yh * zl, pll = yl * zl;
281 
282  /*
283  Proof that 64-bit products can accumulate two more 32-bit values
284  without overflowing:
285 
286  Max 32-bit value is 2^32 - 1.
287  PSum = (2^32-1) * (2^32-1) + (2^32-1) + (2^32-1)
288  = 2^64 - 2^32 - 2^32 + 1 + 2^32 - 1 + 2^32 - 1
289  = 2^64 - 1
290  Therefore it cannot overflow regardless of input.
291  */
292  octa pm = phl + (pll >> 32) + (plh & 0xFFFFFFFF);
293  octa ph = phh + (pm >> 32) + (plh >> 32);
294 
295  *aux = ph;
296 
297  if(ph > 0)
298  *overflow = true;
299 
300  return (pm << 32) | (pll & 0xFFFFFFFF);
301 }
